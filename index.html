An example

You can also get a stronger taste.

type 'a tree =
  Leaf of 'a  |  Node of 'a tree * 'a tree

let rec exists_leaf test = function
  | Leaf v -> test v
  | Node (left, right) ->
      exists_leaf test left
      || exists_leaf test right

let has_even_leaf tree =
  exists_leaf (fun n -> n mod 2 = 0) tree
